streamDisable3D(1)
if(chatClk(owner())){
    OwnerLS=owner():lastSaid():explode(" ")
    if(OwnerLS[1,string]==".play"){
        printColor(vec(0,255,0),"Now Playing: " +OwnerLS[2,string]:toNumber():toString())
        print(SongsList[OwnerLS[2,string]:toNumber(),string])
        print(SongsList[1,string])
        print(SongNames[2,string])
        streamStop(1)
        entity():streamStart(1,SongsList[OwnerLS[2,string]:toNumber(),string],1)
        setName("Now playing" +": " + SongNames[OwnerLS[2,string]:toNumber(),string])
    }
    elseif(OwnerLS[1,string] == ".playowner") {
        owner():streamStart(1,SongsList[OwnerLS[2,string]:toNumber(),string],1)
        printColor(vec(255,0,69),"Now Playing on You: " +OwnerLS[2,string]:toNumber():toString()) 
    }
    elseif(OwnerLS[1,string]==".volume"){
        streamVolume(1,OwnerLS[2,string]:toNumber())
        printColor(vec(120,255,120),"Volume: "+OwnerLS[2,string]:toNumber():toString()+".")
    }
    elseif(OwnerLS[1,string]==".stop"){
        streamStop(1)
        printColor(vec(255,0,0),"All Music Has Been Stopped!")
    }
    elseif(OwnerLS[1,string]==".commands"){
        printColor(vec(0,120,0),"Thank You For Typing .commands. My Current Commands Are:")
        printColor(vec(255,69,255),"1. .play - This plays the song, .play 1-153")
        printColor(vec(255,69,255),"2. .playowner - This plays the song on owner, .playowner 1-153")
        printColor(vec(255,69,255),"3. .volume - This changes the volume, .volume 0-1")
        printColor(vec(255,69,255),"4. .stop - Haults all music")
    }
}
